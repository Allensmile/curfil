785  for (size_t i = 0; i < size; i++) {
786  leftHistogram[i] = leftNode->getHistogram()[i];
787  rightHistogram[i] = rightNode->getHistogram()[i];
788  allHistogram[i] = currentNode->getHistogram()[i];
789  }
790  // assert that actual score is the same as the calculated score
791  double totalLeft = sum(leftNode->getHistogram());
792  double totalRight = sum(rightNode->getHistogram());
793  double actualScore = NormalizedInformationGainScore::calculateScore(size, leftHistogram, rightHistogram,
794  leftRightStride,
795  allHistogram, totalLeft, totalRight);
796  double diff = std::fabs(actualScore - bestSplit.getScore());
797  if (diff > 0.02) {
798  std::ostringstream o;
799  o.precision(10);
800  o << "actual score and best split score differ: " << diff << std::endl;
801  o << "actual score: " << actualScore << std::endl;
802  o << "best split score: " << bestSplit.getScore() << std::endl;
803  o << "total left: " << totalLeft << std::endl;
804  o << "total right: " << totalRight << std::endl;
805  o << "histogram: " << currentNode->getHistogram() << std::endl;
806  o << "histogram left: " << leftNode->getHistogram() << std::endl;
807  o << "histogram right: " << rightNode->getHistogram() << std::endl;
808  throw std::runtime_error(o.str());
809  }

814  /* Train a single random tree breadth-first */
815  void train(FeatureEvaluation& featureEvaluation,
816  RandomSource& randomSource,
817  const std::vector<std::pair<RandomTreePointer, Samples> >& samplesPerNode,
818  int idNode, int currentLevel = 1) const {
819 
820  // Depth exhausted: leaf node
821  if (currentLevel == configuration.getMaxDepth()) {
822  return;
823  }
824 
825  CURFIL_INFO("training level " << currentLevel << ". nodes: " << samplesPerNode.size());
826 
827  utils::Timer trainTimer;
828 
829  std::vector<std::pair<RandomTreePointer, Samples> > samplesPerNodeNextLevel;
830 
831  std::vector<SplitFunction<Instance, FeatureFunction> > bestSplits = featureEvaluation.evaluateBestSplits(
832  randomSource, samplesPerNode);
833 
834  assert(bestSplits.size() == samplesPerNode.size());
835 
836  for (size_t i = 0; i < samplesPerNode.size(); i++) {
837 
838  const std::pair<RandomTreePointer, Samples>& it = samplesPerNode[i];
839 
840  const std::vector<const Instance*>& samples = it.second;
841 
842  boost::shared_ptr<RandomTree<Instance, FeatureFunction> > currentNode = it.first;
843  assert(currentNode);
844 
845  const SplitFunction<Instance, FeatureFunction>& bestSplit = bestSplits[i];
846 
847  // Split all training instances into the subtrees. The way we train
848  // the trees ensures we have no more than 2*N instances to manage in
849  // memory.
850  std::vector<const Instance*> samplesLeft;
851  std::vector<const Instance*> samplesRight;
852 
853  for (size_t sample = 0; sample < samples.size(); sample++) {
854  assert(samples[sample] != NULL);
855  if (bestSplit.split(*samples[sample]) == LEFT) {
856  samplesLeft.push_back(samples[sample]);
857  } else {
858  samplesRight.push_back(samples[sample]);
859  }
860  }
861 
862  assert(samplesLeft.size() + samplesRight.size() == samples.size());
863 
864  boost::shared_ptr<RandomTree<Instance, FeatureFunction> > leftNode = boost::make_shared<RandomTree<Instance,
865  FeatureFunction> >(++idNode, currentNode->getLevel() + 1, samplesLeft, numClasses, currentNode);
866 
867  boost::shared_ptr<RandomTree<Instance, FeatureFunction> > rightNode = boost::make_shared<
868  RandomTree<Instance, FeatureFunction> >(++idNode, currentNode->getLevel() + 1, samplesRight,
869  numClasses, currentNode);
870 
871 #ifndef NDEBUG
872  compareHistograms(currentNode, leftNode, rightNode, bestSplit);
873 #endif
874 
875  if (samplesLeft.empty() || samplesRight.empty()) {
876  CURFIL_ERROR("best split score: " << bestSplit.getScore());
877  CURFIL_ERROR("samples: " << samples.size());
878  CURFIL_ERROR("threshold: " << bestSplit.getThreshold());
879  CURFIL_ERROR("feature: " << bestSplit.getFeature());
880  CURFIL_ERROR("histogram: " << currentNode->getHistogram());
881  CURFIL_ERROR("samplesLeft: " << samplesLeft.size());
882  CURFIL_ERROR("samplesRight: " << samplesRight.size());
883 
884  compareHistograms(currentNode, leftNode, rightNode, bestSplit);
885 
886  if (samplesLeft.empty()) {
887  throw std::runtime_error("no samples in left node");
888  }
889  if (samplesRight.empty()) {
890  throw std::runtime_error("no samples in right node");
891  }
892  }
893 
894  currentNode->addChildren(bestSplit, leftNode, rightNode);
895 
896  if (shouldContinueGrowing(leftNode)) {
897  samplesPerNodeNextLevel.push_back(std::make_pair(leftNode, samplesLeft));
898  }
899 
900  if (shouldContinueGrowing(rightNode)) {
901  samplesPerNodeNextLevel.push_back(std::make_pair(rightNode, samplesRight));
902  }
903  }
904 
905  CURFIL_INFO("training level " << currentLevel << " took " << trainTimer.format(3));
906  if (!samplesPerNodeNextLevel.empty()) {
907  train(featureEvaluation, randomSource, samplesPerNodeNextLevel, idNode, currentLevel + 1);
908  }
909  }
